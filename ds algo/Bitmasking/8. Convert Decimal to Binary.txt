#include <bits/stdc++.h>
using namespace std;

int convertToBinary(int n)
{
	//we will convert the n into is base 10 representation which will look like binary representation of n

	int ans = 0;
	int p = 1;

	while(n>0)
	{
		int last_bit = (n&1);
		ans+= p*last_bit;
		p*=10;
		n=(n>>1);
	}
	return ans;
}

int main()
{
	int n;
	cin>>n;
	cout<<convertToBinary(n);
	return 0;
}
