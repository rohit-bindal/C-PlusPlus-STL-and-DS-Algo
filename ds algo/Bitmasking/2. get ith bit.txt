#include <bits/stdc++.h>
using namespace std;

//task is to find ith bit (i starting from zero and from right side) in a number

int ithBit(int n, int i)
{
	int mask = 1<<i;
	int bit = (n & mask) > 0 ? 1 : 0;

	return bit;
}

int main()
{
	int n,i;
	cin>>n>>i;
    cout<<ithBit(n,i);
	return 0;
}
