#include <bits/stdc++.h>
using namespace std;

int fastExponentiation(int a, int n)
{
  int ans = 1;

  while(n>0)
  {
    if((n&1)==1)
    {
      ans*=a;
    }
    n>>=1; 
    a*=a;
  }
  return ans;
}

int main()
{
  
  int a,n;
  cin>>a>>n;
  cout<<fastExponentiation(a,n);
	return 0;
}
