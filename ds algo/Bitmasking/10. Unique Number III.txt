#include <bits/stdc++.h>
using namespace std;

/*
Every number is occuring thrice in the given array except one. find the unique number
*/

int main()
{
  int count[64] = {0};

  int n;
  cin>>n;
  int no;
  for(int i=0; i<n; i++)
  {
    cin>>no;
    //we will keep adding the bits of upcoming numbers in the count array

    int j=0;
    while(no>0)
    {
      count[j]+= (no&1);
      j++;
      no = no>>1;
    }
  }

  //now lets take mod with 3 of numbers present in the count array 
  int ans = 0;
  int p = 1;
  for(int i=0; i<n; i++)
  {
    count[i] = count[i]%3;
    ans+=count[i]*p;
    p*=2;
  }
  cout<<ans;

	return 0;
}
