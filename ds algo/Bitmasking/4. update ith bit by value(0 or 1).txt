#include <bits/stdc++.h>
using namespace std;

int updateIthBitByValue(int n, int i, int v)
{
   //lets first cleat ith bit
   int mask = (~(1<<i));
   n = (n&mask);

   int newmask = (v<<i);
   n = (n|newmask);

   return n;
}

{
  int n,i,v;
  cin>>n>>i>>v;
