#include <bits/stdc++.h>
using namespace std;

int replaceNbitsByMFromItoJ(int n, int m, int i, int j)
{
  //lets first clear the bits in n from i to j
  int a = (~0)<<(j+1);

  int b = (1<<i) -1;
  int mask = a|b;

  n = n&mask;

  m = (m<<i);

  return n|m;
}

int main()
{
	int n = 15;
	int m = 2;
	int i = 1;
	int j = 3;

	cout<<replaceNbitsByMFromItoJ(n,m,i,j);
	return 0;
}
